1. cron

1.1. 每隔10秒判断端口不存在拉起进程

* * * * * step=10;for (( i=0; i<60; i=(i+step) )); do if ! /usr/sbin/ss -lnt|grep 53088 >/dev/null 2>&1; then cd /inm/app/tengine-V2.1.2-ESB/ && bash ./init.d.tengine restart;fi; sleep $step; done

6. 遍历文件生成md5

#生成隐藏md5文件
find  -type f -name "*.txt"|while read file; do fn=$(echo "${file}"|awk -F'/' '{print $NF}');if [ ! -f ".${fn}.inm" ];then md5sum "$file"|cut -d ' ' -f1> ".${fn}.inm"; fi;  done
find /.-type f -name "*.txt"|while read file; do if [ ! -f "${file}.inm" ]; then md5sum "$file"|cut -d ' ' -f1> "$file".inm; fi;  done

7. 大批量删除文件最佳方式

find /directory/ -type f -not -name 'PATTERN' -print0 | xargs -0 rm

8. SWAP

# 创建 Swap 文件
$ fallocate -l 8G /mnt/swapfile
# 修改权限只有根用户可以访问
$ chmod 600 /mnt/swapfile
# 配置 Swap 文件
$ mkswap /mnt/swapfile
# 开启 Swap
$ swapon /mnt/swapfile

11.1. 操作记录

sudo mkdir /var/log/historymonitor
sudo chmod 003 /var/log/historymonitor
sudo >/var/log/historymonitor/historymonitor.log
sudo chmod 002 /var/log/historymonitor/historymonitor.log
export HISTORY_FILE=/var/log/historymonitor/historymonitor.log
export PROMPT_COMMAND='{ date "+%Y-%m-%d %T ## `hostname` ## $(who am i) ## $realsourceip ## $(history 1 | { read x cmd; echo "$cmd"; })"; } >>$HISTORY_FILE'
realsourceip=`who am i | awk -F"[()]" '{print $2}'`

11.2. ls时间

ls --full-time, 等价于ls -l --time-style=full-iso

11.3. ps时间

ps -o ruser=user1234512345 -e -o pid,ppid,c,lstart,tty,etime,cmd|less

11.4. wget

# 下载整站
wget -r -p -np -k -P ~/tmp/ http://java-er.com
-P 表示下载到哪个目录
-r 表示递归下载
-np 表示不下载旁站连接.
-k 表示将下载的网页里的链接修改为本地链接.
-p 获得所有显示网页所需的元素
-c 断点续传
-nd 递归下载时不创建一层一层的目录，把所有的文件下载到当前目录
-L 递归时不进入其它主机，如wget -c -r www.xxx.org/
-A 指定要下载的文件样式列表，多个样式用逗号分隔
-i 后面跟一个文件，文件内指明要下载的URL
