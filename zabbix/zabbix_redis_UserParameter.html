2.1.1. 自动发现脚本

cat externalscripts/redis_low_discovery.sh
# cat redis_low_discovery.sh
#!/bin/bash
# usage: redis_low_discovery.sh redis
redis() {
    #port=($(sudo ss -4lntp | awk -F "[: ]+" '/redis/ {print $4}'))
    port=($(sudo netstat -tpln | awk -F "[ :]+" '/redis-s/{print $5}'))
    printf '{\n'
    printf '\t"data":[\n'
    for key in ${!port[@]}; do
        if [[ "${#port[@]}" -gt 1 && "${key}" -ne "$((${#port[@]}-1))" ]];then
            socket=`ps aux|grep ${port[${key}]}|grep -v grep|awk -F '=' '{print $10}'|cut -d ' ' -f 1`
            printf '\t {\n'
            printf "\t\t\t\"{#REDISPORT}\":\"${port[${key}]}\"},\n"
        else [[ "${key}" -eq "((${#port[@]}-1))" ]]
            socket=`ps aux|grep ${port[${key}]}|grep -v grep|awk -F '=' '{print $10}'|cut -d ' ' -f 1`
            printf '\t {\n'
            printf "\t\t\t\"{#REDISPORT}\":\"${port[${key}]}\"}\n"
        fi
    done
    printf '\t ]\n'
    printf '}\n'
}

2.1.3. 自动发现规则

  • 自动发现规则

    • 名称：Redis Server Discovery
    • 类型：zabbix客户端(主动式)
    • 键值：redis_low_discovery[redis]
    • 数据更新间隔(秒)：60
    • 保留失去的资源期间(天)：30

  • 过滤器

    • {#REDISPORT}匹配@REDISPORT
    • 名称：REDISPORT
    • 表达式：结果为真 ^([0-9]+)$
