SELECT
    g_name,
    p.host p_name,
    h_name p_h_name,
    h.hostid,
    h.templateName,
    FROM_UNIXTIME(p.lastaccess) lastaccess
FROM
    (SELECT hostid, host, lastaccess FROM hosts WHERE status = 5) p #所有的proxy
    LEFT JOIN
        (SELECT g.name g_name, h.host, h.hostid, h.name h_name, t.templateName
      FROM
         (SELECT h.hostid, ht.templateid, h.name, h.host FROM hosts_templates ht, hosts h WHERE ht.hostid = h.hostid AND h.status=0) h
         INNER JOIN (SELECT hostid, name templateName FROM hosts WHERE status = 3) t ON t.hostid = h.templateid
         INNER JOIN hosts_groups hg ON hg.hostid = h.hostid
         INNER JOIN groups g ON g.groupid = hg.groupid
      WHERE t.templateName = 'Template App Zabbix Proxy') h # 加载了模板的proxy
        ON p.host = h.host
ORDER BY p_name

    from_unixtime( tr.clock, "%Y-%m-%d %H:%i" ) AS date_time,
    g.name AS g_name,
    h.name AS h_name,
    i.key_ AS i_key,
    min( tr.value_min ) AS min_value,
    avg( tr.value_avg ) AS avg_value,

9.1. 告警数量

SELECT g.name g_name, h.name h_name, p.c alert_count
FROM 
    groups g
  JOIN hosts_groups hg ON g.groupid=hg.groupid
  JOIN hosts h ON hg.hostid=h.hostid
  JOIN items i ON h.hostid=i.hostid
  JOIN functions f ON i.itemid=f.itemid
  JOIN triggers r ON f.triggerid=r.triggerid
  JOIN (SELECT objectid, COUNT(*) c FROM problem GROUP BY objectid) p ON r.triggerid=p.objectid
ORDER BY alert_count DESC
