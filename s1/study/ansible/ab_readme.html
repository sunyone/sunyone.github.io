1. 安装

  • 下载tar包
wget https://releases.ansible.com/ansible/ansible-latest.tar.gz
  • 安装
tar -xvf ansible-latest.tar.gz
cd ansible-latest
python setup.py install

mkdir /etc/ansible
cp -p ./examples/ansible.cfg /etc/ansible/.

2. 简单使用

2.1. Ansible 管理机与被管理机做秘钥认证

# 生成密钥
ssh-keygen
# 将公钥写入被管理机
ssh-copy-id -i ~/.ssh/id_rsa.pub -p 22 root@192.168.12.129

2.2. hosts 文件添加被管理机

> /etc/ansible/hosts
vim /etc/ansible/hosts

[nginx_hosts]
192.168.12.129:52222
192.168.12.130:52222

2.3. Ansible 常用模块

2.3.1. ping

ansible nginx_hosts -m ping
ansible nginx_hosts -m ping -o
ansible nginx_hosts -m ping -u ultrainm --ask-pass

# -i          指定 hosts 文件位置
# -u username 指定 SSH 连接的用户名
# -k          指定远程用户密码
# -f          指定并发数
# -s          如需要 root 权限执行时使用 ( 连接用户不是 root 时 )
# -K          -s 时，-K 输入 root 密码

2.3.2. 远程命令模块（ command / script / shell ）

# 远程执行命令
ansible nginx_hosts -a "ls -l"
ansible nginx_hosts -m command -a "free -m"
# 远程执行本地脚本
ansible nginx_hosts -m script -a "/home/test.sh 12 34"
# 执行远程脚本
ansible nginx_hosts -m shell -a "/home/test.sh"

2.3.3. copy 模块

实现主控端向目标主机拷贝文件，类似于 scp 功能

# 向 nginx_hosts 组中主机拷贝 test.sh 到 /tmp 下，属主、组为 root ，权限为 0755
ansible nginx_hosts -m copy -a "src=/home/test.sh desc=/tmp/ owner=root group=root mode=0755"

2.3.4. stat 模块

获取远程文件状态信息，atime/ctime/mtime/md5/uid/gid 等信息

ansible nginx_hosts -m stat -a "path=/etc/syctl.conf"

2.3.5. get_url 模块

实现在远程主机下载指定 URL 到本地，支持 sha256sum 文件校验

ansible nginx_hosts -m get_utl -a "url=http://www.baidu.com dest=/tmp/index.html mode=0440 force=yes"

2.3.6. yum 模块

软件包管理

ansible nginx_hosts -m yum -a "name=curl state=latest"

2.3.7. cron 模块

远程主机 crontab 配置

ansible nginx_hosts -m cron -a "name='check dirs' hour='5,2' job='ls -alh > /dev/null'"

效果：

#Ansible: check dirs
* 5,2 * * * ls -alh > /dev/null

2.3.8. mount 模块

远程主机分区挂载

ansible nginx_hosts -m mount -a "name=/mnt/data src=/dev/sd0 fstype=ext4 opts=ro state=present"

2.3.9. service 模块

远程主机系统服务管理

ansible nginx_hosts -m service -a "name=nginx state=stoped"
ansible nginx_hosts -m service -a "name=nginx state=restarted"
ansible nginx_hosts -m service -a "name=nginx state=reloaded"

2.3.10. user 服务模块

远程主机用户管理

# 添加用户
ansible nginx_hosts -m user -a "name=wang comment='user wang'"
# 删除用户
ansible nginx_hosts -m user -a "name=wang state=absent remove=yes"

3. Ansible-playbook

使用 Ansible-playbook 可以完成一组复杂的动作，例如部署环境、搭建服务、修改配置等。

3.1. 实例：安装zabbix agent

3.1.1. 定义 hosts（ 给哪些主机安装 ）

vim /etc/ansible/hosts

[mini]

129.139.153.78:16283
155.139.190.94:12573

3.1.2. 定义入口文件 install_zabbix_agent.yml

vim /etc/ansible/install_zabbix_agent.yml

- hosts: mini
  roles:
  - install_zabbix_agent

# 可以看到将要安装的主机组为 mini 组，角色为 install_zabbix_agent

3.1.3. 定义角色 install_zabbix_agent

$ tree /etc/ansible/roles/install_zabbix_agent/

├── files
│    └── zabbix-2.4.5.tar.gz
├── tasks
│    └── main.yml
├── templates
│    ├── init.zabbix_agentd.sh
│    └── zabbix_agentd.conf
└── vars
      └── main.yml

## 建立 files     目录，存放编译安装过的 zabbix_agent目录的压缩文件，用于拷贝到远程主机
## 建立 tasks     目录，用于编写将要执行的任务
## 建立 templates 目录，用于存放可变的模板文件
## 建立 vars      目录，用于存放变量信息
  • vars/main.yml

cat /etc/ansible/roles/install_zabbix_agent/vars/main.yml

zabbix_dir: /usr/local
zabbix_version: 2.4.5
zabbix_user: zabbix
zabbix_port: 10050
zabbix_server_ip: 131.142.101.120
  • tasks/main.yml

cat /etc/ansible/roles/install_zabbix_agent/tasks/main.yml

  - name: Install Software
    yum: name={{ item }} state=latest
    with_items:
      - libcurl-devel
  - name: Create Zabbix User
    user: name={{ zabbix_user }} state=present createhome=no shell=/sbin/nologin
  - name: Copy Zabbix.tar.gz
    copy: src=zabbix-{{ zabbix_version }}.tar.gz dest={{ zabbix_dir }}/src/zabbix-{{ zabbix_version }}.tar.gz owner=root group=root
  - name: Uncompression Zabbix.tar.gz
    shell: tar zxf {{ zabbix_dir }}/src/zabbix-{{ zabbix_version }}.tar.gz -C {{ zabbix_dir }}/
  - name: Copy Zabbix Start Script
    template: src=zabbix_agentd dest=/etc/init.d/zabbix_agentd owner=root group=root mode=0755
  - name: Copy Zabbix Config File
    template: src=zabbix_agentd.conf dest={{ zabbix_dir }}/zabbix/etc/zabbix_agentd.conf owner={{ zabbix_user }} group={{ zabbix_user }} mode=0644
  - name: Modify Zabbix Dir Permisson
    file: path={{ zabbix_dir }}/zabbix owner={{ zabbix_user }} group={{ zabbix_user }} mode=0755 recurse=yes
  - name: Start Zabbix Service
    shell: /etc/init.d/zabbix_agentd start
  - name: Add Boot Start Zabbix Service
    shell: chkconfig --level 35 zabbix_agentd on
  • templates/init.zabbix_agentd.sh
# 注意变量引用方式
-- skip --
BASEDIR={{ zabbix_dir }}/zabbix
-- skip --
  • templates/zabbix_agentd.conf
-- skip --
Server={{ zabbix_server_ip }}
-- skip --
ListenPort={{ zabbix_port }}
-- skip --
Hostname={{ ansible_all_ipv4_addresses[1] }}
-- skip --

3.1.4. 执行安装

ansible-playbook /etc/ansible/install_zabbix_agent.yml

3.2. 实例：删除zabbix agent

delete_zabbix_agent.yml

shell > vim /etc/ansible/delete_zabbix_agent.yml

---
- hosts: mini
  roles:
  - delete_zabbix_agent

vim /etc/ansible/roles/delete_zabbix_agent/tasks/main.yml

- name: Stop Zabbix_agent
  shell: pgrep zabbix_agentd | xargs kill
  ignore_errors: yes
- name: Delete Boot Start
  shell: chkconfig --del zabbix_agentd
- name: Delete Start Script
  shell: rm -rf /etc/init.d/zabbix_agentd
- name: Delete Install Dir
  shell: rm -rf {{ zabbix_dir }}/zabbix
- name: Delete Software
  shell: rm -rf {{ zabbix_dir }}/src/zabbix-{{ zabbix_version }}.tar.gz
- name: Delete Log File
  shell: rm -rf /tmp/zabbix_agentd.log
- name: Delete Zabbix User
  user: name={{ zabbix_user }} state=absent remove=yes

vim /etc/ansible/roles/delete_zabbix_agent/vars/main.yml

zabbix_dir: /usr/local
zabbix_version: 2.4.5
zabbix_user: zabbix

4. 附录：hosts配置规则

Ansible 定义主机、组规则的配置文件。默认/etc/ansible/hosts 文件

vim /etc/ansible/hosts

www.abc.com     # 定义域名

192.168.1.100   # 定义 IP

192.168.1.150:37268   # 指定端口号

[WebServer]           # 定义分组

192.168.1.10
192.168.1.20
192.168.1.30

[DBServer]            # 定义多个分组

192.168.1.50
192.168.1.60

Monitor ansible_ssh_port=12378 ansible_ssh_host=192.168.1.200   # 定义别名

# ansible_ssh_host 连接目标主机的地址

# ansible_ssh_port 连接目标主机的端口，默认 22 时无需指定

# ansible_ssh_user 连接目标主机默认用户

# ansible_ssh_pass 连接目标主机默认用户密码

# ansible_ssh_connection 目标主机连接类型，可以是 local 、ssh 或 paramiko

# ansible_ssh_private_key_file 连接目标主机的 ssh 私钥

# ansible_*_interpreter 指定采用非 Python 的其他脚本语言，如 Ruby 、Perl 或其他类似 ansible_python_interpreter 解释器

[webservers]         # 主机名支持正则描述

www[01:50].example.com

[dbservers]

db-[a:f].example.com

## ansible.cfg  此文件为 ansible 的主配置文件
## hosts        用于定义主机组
## roles        定义不同的角色
## install_zabbix_agent.yml  用于安装 zabbix_agent 的引导文件
## delete_zabbix_agent.yml   删除已安装的 zabbix_agent 的引导文件
## file      目录：用于存放将要拷贝到远程主机的安装包等
## tasks     目录：将要执行的所有任务，如果比较复杂，可以单独定义不同的任务，最后在 main.yml 文件中引用即可
## templates 目录：模板目录，这里存放着一些可变的文件，即：每台主机上的这些文件中的内容都不完全相同
## vars      目录：用于存放变量
